\bigbreak \noindent \bigbreak \noindent
\begin{LARGE}
	\textbf{Recall:}
\end{LARGE}
\begin{large}
 \begin{enumerate}
	 \item Exponential functions are one-to-one with H.A. $y=0$ 
	 \item One-to-one functions have inverse functions
 \end{enumerate} 

 \nt{The inverse of exponential functions is the logarithmic function.}
\end{large}
\bigbreak \noindent \bigbreak \noindent
\dfn{}{
Leta be a positive real number different from 1. The logarithm of $\mathbf{x}$ with base a is defined by $\log _a x=y$ if and only if $x=a^y$ for every $x>0$ and every real number $y$. 
}
\bigbreak \noindent \bigbreak \noindent
\begin{large}
	\textbf{Example 1:}
	Change to logarithmic form
\end{large}

\bigbreak \noindent 
\begin{large}
a) $3^{-4}=\frac{1}{81}$
\end{large}
\pf{Soultion}

$$
\log _3 \frac{1}{81}=-4
$$

\bigbreak \noindent \bigbreak \noindent
\begin{large}
b) $(0.9)^t=\frac{1}{2}$
\end{large}

\pf{Solution}

$$
\log _{0.9} \frac{1}{2}=t
$$
\bigbreak \noindent \bigbreak \noindent

\begin{large}
	\noindent \textbf{Example 2:} 
	Change to exponential form
\end{large}
\bigbreak \noindent 
\begin{large}
a) $\log _a \frac{1}{256}=-4$
\end{large}

\pf{Solution}

$$a^{-4} = \frac{1}{256}$$
\bigbreak \noindent
\begin{large}
  b) $\log _a 343=\frac{3}{4}$
\end{large}

\pf{Solution}

$$a^{\frac{3}{4}}=343$$

\bigbreak \noindent \bigbreak \noindent
\begin{large}
	\textbf{Example 3:}
	Find the exact value of each logarithm
\end{large}
\bigbreak \noindent
\begin{large}
  a) $\log _5 \sqrt[3]{25}$
\end{large}

\pf{Solution}

$$
5^x=25^{1 / 3}
$$
$$
5^x=\left(5^2\right)^{1 / 3}
$$
$$
5^x=5^{2 / 3}
$$
$$ \frac{2}{3} \rightarrow solution$$
\bigbreak \noindent
\begin{large}
b) $\log _{\sqrt{3}} 9$
\end{large}

\pf{Solution}

$$(\sqrt{3}^3)=9$$
$$(3^{\frac{1}{2}})^{x}=3^2$$
$$\frac{1}{2}x = 2$$
$$x=4$$

\pagebreak
\dfn{}{
	Common Logarithm: (Logarithm with base 10) $\log x=\log _{10} x$ for every $x>0$
Natural Logarithm: (Logarithm base e) $\ln x=\log _e x$ for every $x>0$
}
\nt{
Properties of logarithms \\
1. $\log _a 1=0$ \\
2. $\log _a a=1$ \\ 
3. $\log _a a^x=x$ \\
4. $a^{\log _a x}=x$
}
